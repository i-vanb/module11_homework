Задание 3.

Написать функцию, которая принимает число как аргумент и возвращает функцию, которая также принимает число как аргумент и возвращает сумму этих двух чисел. Выведите в консоль результат.

Решение

function addNum(n) {
    return function (m) {
        return n + m
    }
}

const func = addNum(3)
console.log(func(8))
